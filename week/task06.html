function r(i) {
  switch (i) {
    case 1:
      $('.5-1').text('Red');
      break;
    case 2:
      $('.5-1').text('Blue');
      break;
    case 3:
      $('.5-1').text('Green');
      break;
    case 4:
      $('.5-1').text('Yellow');
      break;
    case 5:
      $('.5-1').text('Orange');
      break;
    default:
      $('.5-1').text('None');	
      break;
  }
}
